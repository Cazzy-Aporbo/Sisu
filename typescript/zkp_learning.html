zk-SNARKs Example 
const zkSnarks = require('snarkjs');

// Define the circuit for proof computation
const circuit = compile('circuit.circom');

// Setup proving and verification keys
const { proof, publicSignals } = await zkSnarks.groth16.fullProve(
  { "secret": 123, "public": 456 },
  circuit,
  provingKey
);

// Verify the proof
const verified = await zkSnarks.groth16.verify(
  verificationKey,
  publicSignals,
  proof
);

console.log("Proof verified:", verified);
