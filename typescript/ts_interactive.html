// generics.ts - Advanced generic examples
function identity<T>(arg: T): T {
    return arg;
}

function merge<T, U>(obj1: T, obj2: U): T & U {
    return {...obj1, ...obj2};
}

class DataHolder<T> {
    private data: T;
    
    constructor(data: T) {
        this.data = data;
    }
    
    getValue(): T {
        return this.data;
    }
    
    setValue(value: T): void {
        this.data = value;
    }
}

// Constrained generic with default
function getProperty<T, K extends keyof T>(obj: T, key: K): T[K] {
    return obj[key];
}

Generics Demo

Generics allow creating reusable components that can work with multiple types while maintaining type safety.

// Generic function example
function reverse<T>(items: T[]): T[] {
  return items.reverse();
}

// Usage with different types
const numbers = [1, 2, 3, 4, 5];
const reversedNumbers = reverse(numbers);
// reversedNumbers is number[]

const strings = ["a", "b", "c"];
const reversedStrings = reverse(strings);
// reversedStrings is string[]

Generic Constraints

You can constrain generics to types that have certain properties using the extends keyword.

interface HasLength {
  length: number;
}

function logLength<T extends HasLength>(arg: T): T {
  console.log(arg.length);
  return arg;
}

// These will work
logLength([1, 2, 3]);
logLength("hello");
logLength({ length: 10, value: "test" });

// This will cause a compile error
// logLength(42); // Error: number doesn't have a length property

Advanced Types

TypeScript's advanced type features enable powerful type manipulation and transformation.

Conditional Types

Conditional types allow types to be selected based on conditions.

type IsString<T> = T extends string ? true : false;

type T1 = IsString<string>; // true
type T2 = IsString<number>; // false

// Extract type from array
type ExtractArrayType<T> = T extends (infer U)[] ? U : never;

type NumberArray = number[];
type NumberType = ExtractArrayType<NumberArray>; // number
